CREATE TABLE BARISTA(
    BARISTA_ID INTEGER, 
    BARISTA_NAME VARCHAR2(30),
    BARISTA_ADDRESS VARCHAR2(30),
    PHONE_NUM INTEGER,
    BARISTA_SALARY INTEGER,
    PRIMARY KEY (BARISTA_ID)
);

CREATE TABLE COFFEE_SHOP(
    SHOP_ID INTEGER, 
    SHOP_NAME VARCHAR2(30),
    SHOP_ADDRESS VARCHAR2(30),
    PHONE_NUM INTEGER,
    PRIMARY KEY (SHOP_ID)
);

CREATE TABLE SERVER(
    SERVER_ID INTEGER, 
    SERVER_NAME VARCHAR2(30),
    SERVER_ADDRESS VARCHAR2(30),
    PHONE_NUM INTEGER,
    SERVER_SALARY INTEGER,
    PRIMARY KEY (SERVER_ID)
);

CREATE TABLE ITEMS(
    ITEM_ID INTEGER, 
    ITEM_COST INTEGER, 
    QUANTITY INTEGER, 
    ITEM_NAME VARCHAR2(30),
    ITEM_DESCRIPTION VARCHAR2(30),
    PRIMARY KEY (ITEM_ID)
);

CREATE TABLE CUSTOMER(
    CUSTOMER_ID INTEGER, 
    CUSTOMER_NAME VARCHAR2(30),
    PRIMARY KEY (CUSTOMER_ID)
);

CREATE TABLE MANAGER(
    MANAGER_ID INTEGER, 
    NAME VARCHAR2(30),
    MANAGER_ADDRESS VARCHAR2(30),
    PHONE_NUMBER INTEGER,
    MANAGER_SALARY INTEGER,
    PRIMARY KEY (MANAGER_ID)
);

CREATE TABLE BILL(
    BILL_ID INTEGER,
    PRICE INTEGER,
    DECSRIPTION VARCHAR2(30),
    DATE_B DATE,
    ORDER_ID INTEGER,
    PRIMARY KEY (BILL_ID),
    FOREIGN KEY (ORDER_ID) REFERENCES ORDERS (ORDER_ID)
   );

CREATE TABLE ORDER_DETAILS(
    COMPLETED VARCHAR2(30),
    RECEIVED VARCHAR2(30),
    TOTAL_BILL INTEGER,
    ORDER_ID INTEGER,
    FOREIGN KEY (ORDER_ID) REFERENCES ORDERS (ORDER_ID)
   );

CREATE TABLE ORDERS(
    ORDER_ID INTEGER,
    SERVER_ID INTEGER,
    BARISTA_ID INTEGER,
    CUSTOMER_ID INTEGER,
    PRIMARY KEY (ORDER_ID),
    FOREIGN KEY (SERVER_ID) REFERENCES SERVER (SERVER_ID),
    FOREIGN KEY (BARISTA_ID) REFERENCES BARISTA (BARISTA_ID),
    FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER (CUSTOMER_ID)
    );
