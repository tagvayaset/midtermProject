CREATE TABLE BARISTA(
    BARISTA_ID INTEGER, 
    BARISTA_NAME VARCHAR2(30),
    BARISTA_ADDRESS VARCHAR2(30),
    PHONE_NUM INTEGER,
    PRIMARY KEY (BARISTA_ID)
);

CREATE TABLE COFFEE_SHOP(
    SHOP_ID INTEGER, 
    SHOP_NAME VARCHAR2(30),
    SHOP_ADDRESS VARCHAR2(30),
    PHONE_NUM INTEGER,
    PRIMARY KEY (SHOP_ID)
);

CREATE TABLE SERVER(
    SERVER_ID INTEGER, 
    SERVER_NAME VARCHAR2(30),
    SERVER_ADDRESS VARCHAR2(30),
    PHONE_NUM INTEGER,
    PRIMARY KEY (SERVER_ID)
);

CREATE TABLE ITEMS(
    ITEM_ID INTEGER, 
    ITEM_COST INTEGER, 
    QUANTITY INTEGER, 
    ITEM_NAME VARCHAR2(30),
    ITEM_DESCRIPTION VARCHAR2(30),
    PRIMARY KEY (ITEM_ID)
);

CREATE TABLE CUSTOMER(
    CUSTOMER_ID INTEGER, 
    CUSTOMER_NAME VARCHAR2(30),
    PRIMARY KEY (CUSTOMER_ID)
);

CREATE TABLE MANAGER(
    MANAGER_ID INTEGER, 
    NAME VARCHAR2(30),
    MANAGER_ADDRESS VARCHAR2(30),
    PHONE_NUMBER INTEGER,
    PRIMARY KEY (MANAGER_ID)
);
